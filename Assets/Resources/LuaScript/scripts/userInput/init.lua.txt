---@class UserInputHandle
---处理战斗系统中所有的玩家输入
local M = {}

---@type IUserInput
M.handle = nil

function M.init()
    ---如果是在线模式
    if lua and lua.onlineMode then
        M.handle = require"scripts.userInput.UserInputOnline"
    else
        M.handle = require"scripts.userInput.UserInputOffline"
    end
end

return M
