---lua端对mvc View的基本实现，相当于抽象类
---@class mvc.View
local View = class("View")
---@type UnityEngine.GameObject
View.go = nil
View.mvcName = ""
View.list = {}

---构造函数
function View:ctor(data)
    
end

---@param go UnityEngine.GameObject
function View:BindGameObject(go)
    if self.go == nil then
        self.go = go
    end
end

---@param mono UnityEngine.MonoBehaviour
function View:BindMono(mono)
    if self.go == nil then
        self.go = mono.gameObject
    end
end

---@return boolean
function View:Bind()
    return self.go ~= nil
end

---@return UnityEngine.GameObject
function View:GetBindGameObject()
    return self.go
end

---@return string
function View:GetMvcName()
    return self.mvcName
end
