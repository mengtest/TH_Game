function ConfirmButton()
    local name = nameInput.text
    local pwd = pwdInput.text

    -- print(self.gameObject)

    if #name == 0 then
        CS.Global.Log("用户名不能为空")
        return
    end

    if #pwd == 0 then
        CS.Global.Log("请输入密码")
        return
    end

    CS.Global.Log(name)
    CS.Global.Log(pwd)
    CS.Net.NetHelper.GetInstance():Login(name, pwd, "loginCallback")
end

function loginCallback(a, b, c)
    -- local go = self.gameObject
    -- go:Destroy(go)
    -- do
    --     return
    -- end

    -- CS.Global.Log(self.name)
    if a == true then
        CS.Global.Log("登入成功！")
        self.gameObject:SetActive(false)
        -- CS.Global.NavigateTo("MainScene", true)
        
        for k, s in ipairs(CS.UnityEngine.SceneManagement.SceneManager.GetActiveScene():GetRootGameObjects()) do
            if s.name == "Canvas" then
                s:SetActive(false)
            end
        end
    else 
        CS.Global.Log(c)
    end
end

function CancelButton()
    CS.UnityEngine.Object.Destroy(self)
end
