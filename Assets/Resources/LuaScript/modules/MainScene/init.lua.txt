local uiList = {
    LeftTopUI = LeftTopUI,
    RightTopUI = RightTopUI,
    LeftUI = LeftUI,
    RightUI = RightUI,
    BottomUI = BottomUI
}

---@param this LuaFramework.LuaManager
function init(this)
    local canvas = CS.Global.GetCurCanvas()
    for i = 0, canvas.transform.childCount - 1 do
        local child = canvas.transform:GetChild(i)
        local f = uiList[child.name]
        if f ~= nil then
            f(child)
        end
    end
end

function ExpBar(this)
    ---@type UnityEngine.GameObject
    local go = this
    local exp = go:GetComponent(typeof(CS.UnityEngine.UI.Slider))
    exp.value = 50
end

function MainQuest(this)
    ---@type UnityEngine.GameObject
    local go = this
    local btn = go:GetComponent(typeof(CS.UnityEngine.UI.Button))
    btn.onClick:AddListener(function()
        CS.Global.NavigateTo("StoryScene", false)
    end)
end
